public enum KeyType {
    PADLOCK(1024),

    BUTTON(10000),

    DIAL(30000),

    FINGER(1000000);

    private final int limit;

    KeyType(int limit) {
        this.limit = limit;
    }

    public int getLimit() {
        return limit;
    }

}

public class StrongBox<T> {


    private T data;
    private KeyType keyType = KeyType.PADLOCK;
    private int usageNumber;

    public StrongBox(KeyType keyType) {
        this.keyType = keyType;
    }

    public void put(T data) {
        this.data = data;

    }


    public T get() {

        return data;

    }
}

class StrongBoxTest {
    @Test
    void StrongBoxTest() {
        //given

        StrongBox<String> box = new StrongBox<>(KeyType.PADLOCK);


        box.put("보물");

        for (int i = 0; i < 1023; i++) {
            assertNull(box.get());
        }

        assertEquals("보물", box.get());

    }

}
