## 오늘 배운 내용

- 정규 표현식 : 특별한 규칙을 가진 문자열의 집합을 표현는데 사용하는 형식언어. (많은 텍스트 편집기와 프로그래밍 언어에서 문자열의 검색 및 치환을 위해 지원)

## 기억할 것

- substring(,) : 문자열의 일부를 떼어낼수 있다.
- replace(" "," ") : 문자열 일부를 치환 할 수 있다.
- split(" ") : 문자열을 분리 가능
- toLowerCase() : 대문자를 소문자로 변경 가능
- indexOf(' ') : 그 문자가 몇 번째 위치에 있는지 확인 할 수 있다.
- length() : 문자열 길이 확인
- isEmpty() : 길이가 0인지
- contains() : 포함 관계
- endsWith() : 끝나는 단어가 맞는지
- lastIndexOf() : 뒤에서 몇 번째에 단어가 있는지
- append() 메서드로 결합한 결과를 내부 메모리(버퍼)에 담아 두고 toString()으로 결과를 얻음
- 정규 표현식
- .은 임의의 한 문자
- []는 괄호 안의 문자중 하나를 랜덤으로 출력
- ^가 있을 땐 그 문자 출력을 제외
- A-Z 는 A부터 Z의 값을 랜덤 출력
- ^는 맨앞
- $는 맨뒤
- {n}는 n의 문자가 있을때 앞문자를 n번 반복한다는 뜻
- {n,}는 n번 이상 반복
- {n,m}는 n번 이상 m번 이하 반복
- ? 는 앞 글자 0번 또는 1번
- "+" 는 앞 글자 1번 이상 반복

```java
//문자열 처리 결합 예시
void plus() {
    String s1 = "Hello";
    String s2 = " Java";
    System.out.println(s1 + s2);
}


// 문자열 일부 뗴어내기
void substring() {
    String s1 = "Hello";

    System.out.println(s1.substring(0, 2));
}

//문자열 일부 치환
void replace() {
    String s1 = "Hello";

    System.out.println(s1.replace("LL", "XX"));
}

//문자열 분리
void split() {
    String s1 = "1,2,3";

    String splited[] = s1.split(","); //,을 기점으로 분리

    for (
            String s : splited) {

        System.out.println(s);
    }
}

//문자 변환
void StringHiLo() {
    String s1 = "Hello";

    System.out.println(s1.toLowerCase()); //소문자로
    System.out.println(s1.toUpperCase()); //대문자로
    System.out.println(s1.trim()); //좌우 공백 제거
    System.out.println(s1.replace("ell", "al")); // Hello에서 halo로
}

//문자열 길이
void lengthInfo() {
    String s1 = "Java";
    System.out.println(s1.length());
    System.out.println(s1.isEmpty());
}

//문자열 검색
void stirngSearch() {
    String s1 = "Java and JavaScript";
    System.out.println(s1.contains("Java")); //true
    System.out.println(s1.endsWith("Java")); //false
    System.out.println(s1.indexOf("Java")); //0
    System.out.println(s1.lastIndexOf("Java")); //9
}

//StringBuilder
void StringBuilder() {
    StringBuilder sb = new StringBuilder();
    for (int i = 0; i < 10000; i++) {
        sb.append("Java"); //결합
    }
    String s = sb.toString(); //완성된 결과
}

//메서드 체인

void methodChain() {
    StringBuilder sb = new StringBuilder();
    sb.append("Hello").append("Java").append("World");
}
```

```java
public class Main {
    public static void main(String[] args) {

        StringBuilder num = new StringBuilder();

        for (int i = 1; i < 101; i++) {
            num.append(i);
            if (i < 100) {
                num.append(",");
            }
        }

        java.lang.String[] a = num.toString().split(",");
        for (java.lang.String word : a) {
            System.out.println(word + "->");
        }

        //
        System.out.println("^A\\d[\\d^a-z]");
        System.out.println("^U[a-z^0-9{3}]");


    }

    java.lang.String getPath(java.lang.String path, java.lang.String fileName) {
        StringBuilder thePath = new StringBuilder();
        if (!path.endsWith("\\")) {
            thePath.append("\\");
        }
        thePath.append(path);
        thePath.append(fileName);
        return thePath.toString();
    }


}
```
